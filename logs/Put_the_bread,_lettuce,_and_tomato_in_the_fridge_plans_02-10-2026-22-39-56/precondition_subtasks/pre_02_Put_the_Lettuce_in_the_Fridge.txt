# Task Description: Put the Lettuce in the Fridge.

# GENERAL TASK DECOMPOSITION 
# Decompose and parallel subtasks where ever possible
# Independent subtasks:

# Initial condition analyze due to previous subtask:
#1. Robot not at lettuce location
#2. Robot not holding lettuce
#3. Fridge is initially closed

# SubTask 2: Put the Lettuce in the Fridge. 
    Skills Required: GoToObject, PickupObject, OpenObject, PutObjectInFridge, CloseObject
    Related Objects: lettuce(Location=CounterTop), fridge(Location=Floor)

GoToObject(robot1, lettuce)
   Preconditions:
     (not (inaction robot1))
   Effects:
     (at robot1 lettuce)

PickupObject(robot1, lettuce, CounterTop)
   Preconditions:
     (at-location lettuce CounterTop)
     (at robot1 lettuce)
     (not (inaction robot1))
   Effects:
     (holding robot1 lettuce)

GoToObject(robot1, fridge)
   Preconditions:
     (not (inaction robot1))
   Effects:
     (at robot1 fridge)

OpenObject(robot1, fridge)
   Preconditions:
     (not (inaction robot1))
     (at robot1 fridge)
   Effects:
     (object-open robot1 fridge)
     (fridge-open fridge)

PutObjectInFridge(robot1, lettuce, fridge)
   Preconditions:
     (holding robot1 lettuce)
     (at robot1 fridge)
     (not (inaction robot1))
     (is-fridge fridge)
     (fridge-open fridge)
   Effects:
     (at-location lettuce fridge)
     (not (holding robot1 lettuce))
     (not (inaction robot1))

CloseObject(robot1, fridge)
   Preconditions:
     (not (inaction robot1))
     (at robot1 fridge)
   Effects:
     (object-close robot1 fridge)
     (not (fridge-open fridge))

Goal condition: 
(at-location lettuce fridge)
