# SubTask 2: Extinguish CanyonFire using Sand.
    Skills Required: GoToObject, GetSupply, UseSupply
    Related Objects: ReservoirSierra(type=Sand), CanyonFire_Region_1(fire=CanyonFire)

GoToObject(robot1, ReservoirSierra)
   Preconditions:
     (not (inaction robot1))
   Effects:
     (at robot1 ReservoirSierra)

GetSupply(robot1, ReservoirSierra, Sand)
   Preconditions:
     (at robot1 ReservoirSierra)
     (has-resource ReservoirSierra Sand)
     (is-supply Sand)
   Effects:
     (has-supply robot1 Sand)

GoToObject(robot1, CanyonFire_Region_1)
   Preconditions:
     (not (inaction robot1))
   Effects:
     (at robot1 CanyonFire_Region_1)

UseSupply(robot1, CanyonFire_Region_1, Sand)
   Preconditions:
     (at robot1 CanyonFire_Region_1)
     (has-supply robot1 Sand)
     (is-region CanyonFire_Region_1)
     (fire-active CanyonFire_Region_1)
     (region-of CanyonFire_Region_1 CanyonFire)
     (supply-for-fire Sand CanyonFire)
   Effects:
     (not (fire-active CanyonFire_Region_1))
     (not (has-supply robot1 Sand))

Goal condition:
(not (fire-active CanyonFire_Region_1))
