# SubTask 5: Extinguish PlainFire using Water.
    Skills Required: GoToObject, GetSupply, UseSupply
    Related Objects: ReservoirDelta(type=Water), PlainFire_Region_1(fire=PlainFire)

GoToObject(robot1, ReservoirDelta)
   Preconditions:
     (not (at robot1 ReservoirDelta))
   Effects:
     (at robot1 ReservoirDelta)

GetSupply(robot1, ReservoirDelta, Water)
   Preconditions:
     (at robot1 ReservoirDelta)
     (has-resource ReservoirDelta Water)
     (is-supply Water)
   Effects:
     (has-supply robot1 Water)

GoToObject(robot1, PlainFire_Region_1)
   Preconditions:
     (not (at robot1 PlainFire_Region_1))
   Effects:
     (at robot1 PlainFire_Region_1)

UseSupply(robot1, PlainFire_Region_1, Water)
   Preconditions:
     (at robot1 PlainFire_Region_1)
     (has-supply robot1 Water)
     (is-region PlainFire_Region_1)
     (fire-active PlainFire_Region_1)
     (exists (?f - object)
       (and
         (region-of PlainFire_Region_1 ?f)
         (supply-for-fire Water ?f)
       )
     )
   Effects:
     (not (fire-active PlainFire_Region_1))
     (not (has-supply robot1 Water))

Goal condition:
(not (fire-active PlainFire_Region_1))
