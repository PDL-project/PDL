# SubTask 9: Find and rescue LostPersonMaya. [Multi-Agent: 2 robots required for Carry+DropOff]
    Skills Required: Explore, GoToObject, Carry, DropOff
    Related Objects: LostPersonMaya(type=Person), DepositFacility(type=Deposit)
    Note: Moving a person requires 2 agents collaborating simultaneously.

Explore(robot1)
   Preconditions:
     (not (person-found LostPersonMaya))
   Effects:
     (person-found LostPersonMaya)

GoToObject(robot1, LostPersonMaya)
   Preconditions:
     (person-found LostPersonMaya)
   Effects:
     (at robot1 LostPersonMaya)

GoToObject(robot2, LostPersonMaya)
   Preconditions:
     (person-found LostPersonMaya)
   Effects:
     (at robot2 LostPersonMaya)

Carry(robot1, LostPersonMaya)
   Preconditions:
     (at robot1 LostPersonMaya)
     (person-found LostPersonMaya)
     (not (carrying robot1 LostPersonMaya))
   Effects:
     (carrying robot1 LostPersonMaya)

Carry(robot2, LostPersonMaya)
   Preconditions:
     (at robot2 LostPersonMaya)
     (person-found LostPersonMaya)
     (not (carrying robot2 LostPersonMaya))
   Effects:
     (carrying robot2 LostPersonMaya)

GoToObject(robot1, DepositFacility)
   Preconditions:
     (carrying robot1 LostPersonMaya)
   Effects:
     (at robot1 DepositFacility)

GoToObject(robot2, DepositFacility)
   Preconditions:
     (carrying robot2 LostPersonMaya)
   Effects:
     (at robot2 DepositFacility)

DropOff(robot1, LostPersonMaya, DepositFacility)
   Preconditions:
     (at robot1 DepositFacility)
     (carrying robot1 LostPersonMaya)
   Effects:
     (not (carrying robot1 LostPersonMaya))
     (person-rescued LostPersonMaya)

DropOff(robot2, LostPersonMaya, DepositFacility)
   Preconditions:
     (at robot2 DepositFacility)
     (carrying robot2 LostPersonMaya)
   Effects:
     (not (carrying robot2 LostPersonMaya))
     (person-rescued LostPersonMaya)

Goal condition:
(person-rescued LostPersonMaya)
