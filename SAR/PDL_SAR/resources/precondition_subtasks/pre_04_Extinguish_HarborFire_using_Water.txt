# SubTask 4: Extinguish HarborFire using Water.
    Skills Required: GoToObject, GetSupply, UseSupply
    Related Objects: ReservoirDelta(type=Water), HarborFire_Region_1(fire=HarborFire)

GoToObject(robot1, ReservoirDelta)
   Preconditions:
     (not (inaction robot1))
   Effects:
     (at robot1 ReservoirDelta)

GetSupply(robot1, ReservoirDelta, Water)
   Preconditions:
     (at robot1 ReservoirDelta)
     (has-resource ReservoirDelta Water)
     (is-supply Water)
   Effects:
     (has-supply robot1 Water)

GoToObject(robot1, HarborFire_Region_1)
   Preconditions:
     (not (inaction robot1))
   Effects:
     (at robot1 HarborFire_Region_1)

UseSupply(robot1, HarborFire_Region_1, Water)
   Preconditions:
     (at robot1 HarborFire_Region_1)
     (has-supply robot1 Water)
     (is-region HarborFire_Region_1)
     (fire-active HarborFire_Region_1)
     (exists (?f - object)
       (and
         (region-of HarborFire_Region_1 ?f)
         (supply-for-fire Water ?f)
       )
     )
   Effects:
     (not (fire-active HarborFire_Region_1))
     (not (has-supply robot1 Water))

Goal condition:
(not (fire-active HarborFire_Region_1))
