# SubTask 1: Extinguish RidgeFire using Sand.
    Skills Required: GoToObject, GetSupply, UseSupply
    Related Objects: ReservoirSierra(type=Sand), RidgeFire_Region_1(fire=RidgeFire)

GoToObject(robot1, ReservoirSierra)
   Preconditions:
     (not (inaction robot1))
   Effects:
     (at robot1 ReservoirSierra)

GetSupply(robot1, ReservoirSierra, Sand)
   Preconditions:
     (at robot1 ReservoirSierra)
     (has-resource ReservoirSierra Sand)
     (is-supply Sand)
   Effects:
     (has-supply robot1 Sand)

GoToObject(robot1, RidgeFire_Region_1)
   Preconditions:
     (not (inaction robot1))
   Effects:
     (at robot1 RidgeFire_Region_1)

UseSupply(robot1, RidgeFire_Region_1, Sand)
   Preconditions:
     (at robot1 RidgeFire_Region_1)
     (has-supply robot1 Sand)
     (is-region RidgeFire_Region_1)
     (fire-active RidgeFire_Region_1)
     (exists (?f - object)
       (and
         (region-of RidgeFire_Region_1 ?f)
         (supply-for-fire Sand ?f)
       )
     )
   Effects:
     (not (fire-active RidgeFire_Region_1))
     (not (has-supply robot1 Sand))

Goal condition:
(not (fire-active RidgeFire_Region_1))
