# SubTask 9: Move the Tomato to the Fridge.
    Skills Required: GoToObject, PickupObject, PutObjectInFridge
    Related Objects: Tomato(Location=DiningTable), Fridge(Location=Floor)

GoToObject(robot1, tomato)
   Preconditions:
     (not (inaction robot1))
   Effects:
     (at robot1 tomato)

PickupObject(robot1, tomato, diningTable)
   Preconditions:
     (at-location tomato diningTable)
     (at robot1 tomato)
     (not (inaction robot1))
   Effects:
     (holding robot1 tomato)

GoToObject(robot1, fridge)
   Preconditions:
     (not (inaction robot1))
   Effects:
     (at robot1 fridge)

PutObjectInFridge(robot1, tomato, fridge)
   Preconditions:
     (holding robot1 tomato)
     (at robot1 fridge)
     (not (inaction robot1))
     (is-fridge fridge)
     (fridge-open fridge)
   Effects:
     (at-location tomato fridge)
     (not (holding robot1 tomato))
     (not (inaction robot1))
