# SubTask 1: Store the Fork. 
    Skills Required: GoToObject, PickupObject, OpenObject, PutObject, CloseObject
    Related Objects: fork(Location=counterTop), drawer(Location=floor)

GoToObject(robot1, fork)
   Preconditions:
     (not (inaction robot1))
   Effects:
     (at robot1 fork)

PickupObject(robot1, fork, counterTop)
   Preconditions:
     (at-location fork counterTop)
     (at robot1 fork)
     (not (inaction robot1))
   Effects:
     (holding robot1 fork)

GoToObject(robot1, drawer)
   Preconditions:
     (not (inaction robot1))
   Effects:
     (at robot1 drawer)

OpenObject(robot1, drawer)
   Preconditions:
     (not (inaction robot1))
     (at robot1 drawer)
   Effects:
     (object-open robot1 drawer)
     (not (object-close robot1 drawer))

PutObject(robot1, fork, drawer)
   Preconditions:
     (holding robot1 fork)
     (at robot1 drawer)
     (not (inaction robot1))
     (not (object-close robot1 drawer))
   Effects:
     (at-location fork drawer)
     (not (holding robot1 fork))

CloseObject(robot1, drawer)
   Preconditions:
     (not (inaction robot1))
     (at robot1 drawer)
     (object-open robot1 drawer)
   Effects:
     (object-close robot1 drawer)
     (not (object-open robot1 drawer))

Goal condition: 
(at-location fork drawer)

# Task Store the Fork is done.
