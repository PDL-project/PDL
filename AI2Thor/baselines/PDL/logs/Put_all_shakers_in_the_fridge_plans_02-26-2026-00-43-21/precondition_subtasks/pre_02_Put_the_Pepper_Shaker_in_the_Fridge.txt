# SubTask 2: Put the Pepper Shaker in the Fridge. 
    Skills Required: GoToObject, PickupObject, OpenObject, PutObject, CloseObject
    Related Objects: PepperShaker(Location=CounterTop), Fridge(Location=Floor)

GoToObject(robot1, PepperShaker)
   Preconditions:
     (not (inaction robot1))
   Effects:
     (at robot1 PepperShaker)

PickupObject(robot1, PepperShaker, CounterTop)
   Preconditions:
     (at-location PepperShaker CounterTop)
     (at robot1 PepperShaker)
     (not (inaction robot1))
   Effects:
     (holding robot1 PepperShaker)

GoToObject(robot1, Fridge)
   Preconditions:
     (not (inaction robot1))
   Effects:
     (at robot1 Fridge)

OpenFridge(robot1, Fridge)
   Preconditions:
     (not (inaction robot1))
     (at robot1 Fridge)
     (is-fridge Fridge)
   Effects:
     (object-open robot1 Fridge)
     (fridge-open Fridge)

PutObjectInFridge(robot1, PepperShaker, Fridge)
   Preconditions:
     (holding robot1 PepperShaker)
     (at robot1 Fridge)
     (not (inaction robot1))
     (is-fridge Fridge)
     (fridge-open Fridge)
   Effects:
     (at-location PepperShaker Fridge)
     (not (holding robot1 PepperShaker))
     (not (inaction robot1))

CloseFridge(robot1, Fridge)
   Preconditions:
     (not (inaction robot1))
     (at robot1 Fridge)
     (object-open robot1 Fridge)
     (is-fridge Fridge)
   Effects:
     (object-close robot1 Fridge)
     (not (fridge-open Fridge))

Goal condition: 
(at-location PepperShaker Fridge)
