# Initial condition analyze due to previous subtask:
# 1. Robot not at tomato location
# 2. Robot not holding tomato
# 3. Fridge is initially closed

# SubTask 3: Put the Tomato in the Fridge.
    Skills Required: GoToObject, PickupObject, OpenObject, PutObject, CloseObject
    Related Objects: tomato(Location=counterTop), fridge(Location=floor)

GoToObject(robot1, tomato)
   Preconditions:
     (not (inaction robot1))
   Effects:
     (at robot1 tomato)

PickupObject(robot1, tomato, counterTop)
   Preconditions:
     (at-location tomato counterTop)
     (at robot1 tomato)
     (not (inaction robot1))
   Effects:
     (holding robot1 tomato)

GoToObject(robot1, fridge)
   Preconditions:
     (not (inaction robot1))
   Effects:
     (at robot1 fridge)

OpenObject(robot1, fridge)
   Preconditions:
     (not (inaction robot1))
     (at robot1 fridge)
   Effects:
     (object-open robot1 fridge)
     (fridge-open fridge)

PutObject(robot1, tomato, fridge)
   Preconditions:
     (holding robot1 tomato)
     (at robot1 fridge)
     (not (inaction robot1))
     (fridge-open fridge)
   Effects:
     (at-location tomato fridge)
     (not (holding robot1 tomato))

CloseObject(robot1, fridge)
   Preconditions:
     (not (inaction robot1))
     (at robot1 fridge)
     (object-open robot1 fridge)
   Effects:
     (object-close robot1 fridge)
     (not (fridge-open fridge))

Goal condition: 
(at-location tomato fridge)
