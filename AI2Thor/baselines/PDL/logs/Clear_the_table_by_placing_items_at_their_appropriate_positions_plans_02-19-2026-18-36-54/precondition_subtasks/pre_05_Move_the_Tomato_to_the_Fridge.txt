# SubTask 5: Move the Tomato to the Fridge.
    Skills Required: GoToObject, PickupObject, OpenObject, PutObject, CloseObject
    Related Objects: Tomato(Location=DiningTable), Fridge(Location=Floor)

GoToObject(robot1, Tomato)
   Preconditions:
     (not (inaction robot1))
   Effects:
     (at robot1 Tomato)

PickupObject(robot1, Tomato, DiningTable)
   Preconditions:
     (at-location Tomato DiningTable)
     (at robot1 Tomato)
     (not (inaction robot1))
   Effects:
     (holding robot1 Tomato)

GoToObject(robot1, Fridge)
   Preconditions:
     (not (inaction robot1))
   Effects:
     (at robot1 Fridge)

OpenObject(robot1, Fridge)
   Preconditions:
     (not (inaction robot1))
     (at robot1 Fridge)
   Effects:
     (object-open robot1 Fridge)
     (fridge-open Fridge)

PutObject(robot1, Tomato, Fridge)
   Preconditions:
     (holding robot1 Tomato)
     (at robot1 Fridge)
     (not (inaction robot1))
     (fridge-open Fridge)
   Effects:
     (at-location Tomato Fridge)
     (not (holding robot1 Tomato))

CloseObject(robot1, Fridge)
   Preconditions:
     (not (inaction robot1))
     (at robot1 Fridge)
     (object-open robot1 Fridge)
   Effects:
     (object-close robot1 Fridge)
     (not (fridge-open Fridge))

Goal condition: 
(at-location Tomato Fridge)
