# SubTask 5: Place the Lettuce back in the Fridge.
    Skills Required: GoToObject, PickupObject, OpenObject, PutObject, CloseObject
    Related Objects: Lettuce(Location=DiningTable), Fridge(Location=Floor)

GoToObject(robot1, lettuce)
   Preconditions:
     (not (inaction robot1))
   Effects:
     (at robot1 lettuce)

PickupObject(robot1, lettuce, diningTable)
   Preconditions:
     (at-location lettuce diningTable)
     (at robot1 lettuce)
     (not (inaction robot1))
   Effects:
     (holding robot1 lettuce)

GoToObject(robot1, fridge)
   Preconditions:
     (not (inaction robot1))
   Effects:
     (at robot1 fridge)

OpenFridge(robot1, fridge)
   Preconditions:
     (not (inaction robot1))
     (at robot1 fridge)
     (is-fridge fridge)
   Effects:
     (object-open robot1 fridge)
     (fridge-open fridge)

PutObjectInFridge(robot1, lettuce, fridge)
   Preconditions:
     (holding robot1 lettuce)
     (at robot1 fridge)
     (not (inaction robot1))
     (is-fridge fridge)
     (fridge-open fridge)
   Effects:
     (at-location lettuce fridge)
     (not (holding robot1 lettuce))
     (not (inaction robot1))

CloseFridge(robot1, fridge)
   Preconditions:
     (not (inaction robot1))
     (at robot1 fridge)
     (object-open robot1 fridge)
     (is-fridge fridge)
   Effects:
     (object-close robot1 fridge)
     (not (fridge-open fridge))

Goal condition: 
(at-location lettuce fridge)
