# SubTask 6: Place the Tomato back in the Fridge.
    Skills Required: GoToObject, PickupObject, OpenObject, PutObject, CloseObject
    Related Objects: Tomato(Location=DiningTable), Fridge(Location=Floor)

GoToObject(robot1, tomato)
   Preconditions:
     (not (inaction robot1))
   Effects:
     (at robot1 tomato)

PickupObject(robot1, tomato, diningTable)
   Preconditions:
     (at-location tomato diningTable)
     (at robot1 tomato)
     (not (inaction robot1))
   Effects:
     (holding robot1 tomato)

GoToObject(robot1, fridge)
   Preconditions:
     (not (inaction robot1))
   Effects:
     (at robot1 fridge)

OpenFridge(robot1, fridge)
   Preconditions:
     (not (inaction robot1))
     (at robot1 fridge)
     (is-fridge fridge)
   Effects:
     (object-open robot1 fridge)
     (fridge-open fridge)

PutObjectInFridge(robot1, tomato, fridge)
   Preconditions:
     (holding robot1 tomato)
     (at robot1 fridge)
     (not (inaction robot1))
     (is-fridge fridge)
     (fridge-open fridge)
   Effects:
     (at-location tomato fridge)
     (not (holding robot1 tomato))
     (not (inaction robot1))

CloseFridge(robot1, fridge)
   Preconditions:
     (not (inaction robot1))
     (at robot1 fridge)
     (object-open robot1 fridge)
     (is-fridge fridge)
   Effects:
     (object-close robot1 fridge)
     (not (fridge-open fridge))

Goal condition:
(at-location tomato fridge)

# Task Place the Tomato back in the Fridge is done.
