PickupObject(robot1, Spoon, DiningTable)

     (at-location Spoon DiningTable)

   Preconditions:

     (not (object-close robot1 Drawer))

PutObject(robot1, Spoon, Drawer)

     (not (object-close robot1 Drawer))

     (at-location Spoon Drawer)
     (not (holding robot1 Spoon))

CloseObject(robot1, Drawer)
   Preconditions:
     (not (inaction robot1))
