# SubTask 6: Place the Pan into the Cabinet.
    Skills Required: GoToObject, PickupObject, OpenObject, PutObject, CloseObject
    Related Objects: Pan(Location=DiningTable), Cabinet(Location=Floor)

GoToObject(robot1, pan)
   Preconditions:
     (not (inaction robot1))
   Effects:
     (at robot1 pan)

PickupObject(robot1, pan, diningTable)
   Preconditions:
     (at-location pan diningTable)
     (at robot1 pan)
     (not (inaction robot1))
   Effects:
     (holding robot1 pan)

GoToObject(robot1, cabinet)
   Preconditions:
     (not (inaction robot1))
   Effects:
     (at robot1 cabinet)

OpenObject(robot1, cabinet)
   Preconditions:
     (not (inaction robot1))
     (at robot1 cabinet)
   Effects:
     (object-open robot1 cabinet)
     (not (object-close robot1 cabinet))

PutObject(robot1, pan, cabinet)
   Preconditions:
     (holding robot1 pan)
     (at robot1 cabinet)
     (not (inaction robot1))
     (not (object-close robot1 cabinet))
   Effects:
     (at-location pan cabinet)
     (not (holding robot1 pan))

CloseObject(robot1, cabinet)
   Preconditions:
     (not (inaction robot1))
     (at robot1 cabinet)
     (object-open robot1 cabinet)
   Effects:
     (object-close robot1 cabinet)
     (not (object-open robot1 cabinet))

Goal condition:
