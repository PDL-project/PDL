# Initial condition analysis due to previous subtask:

# SubTask 3: Put the Bread in the Fridge.
    Skills Required: GoToObject, PickupObject, OpenObject, PutObjectInFridge, CloseObject
    Related Objects: Bread(Location=CounterTop), Fridge(Location=Floor)

GoToObject(robot1, Bread)
   Preconditions:
     (not (inaction robot1))
   Effects:
     (at robot1 Bread)

PickupObject(robot1, Bread, CounterTop)
   Preconditions:
     (at-location Bread CounterTop)
     (at robot1 Bread)
     (not (inaction robot1))
   Effects:
     (holding robot1 Bread)

GoToObject(robot1, Fridge)
   Preconditions:
     (not (inaction robot1))
   Effects:
     (at robot1 Fridge)

OpenObject(robot1, Fridge)
   Preconditions:
     (not (inaction robot1))
     (at robot1 Fridge)
   Effects:
     (object-open robot1 Fridge)
     (fridge-open Fridge)

PutObjectInFridge(robot1, Bread, Fridge)
   Preconditions:
     (holding robot1 Bread)
     (at robot1 Fridge)
     (not (inaction robot1))
     (is-fridge Fridge)
     (fridge-open Fridge)
   Effects:
     (at-location Bread Fridge)
     (not (holding robot1 Bread))
     (not (inaction robot1))

CloseObject(robot1, Fridge)
   Preconditions:
     (not (inaction robot1))
     (at robot1 Fridge)
     (object-open robot1 Fridge)
   Effects:
     (object-close robot1 Fridge)
     (not (fridge-open Fridge))

Goal condition: 
(at-location Bread Fridge)
